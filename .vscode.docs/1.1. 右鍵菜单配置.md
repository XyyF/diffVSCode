### package.json文件配置
```js
// package.json
{
    // 声明【事件】
    "activationEvents": [
		"onCommand:elfin.wcn.retrofitJs"
    ],
    "contributes": {
        // 全局注册【命令】
		"commands": [
			{
                // 命令对应的事件
                "command": "elfin.wcn.retrofitJs",
                // 命令展示的名称
                "title": "[未成年]格式化Js文件"
            }
        ],
        // 菜单 配置内容
        "menus": {
            // 菜单-编辑器右键菜单 配置内容
            "editor/context": [
                {
                    // 配置内容出现条件
                    "when": "editorFocus && resourceExtname == .js && resourcePath =~ /xiaochengxu-teenager/",
                    // 菜单项对应的事件
                    "command": "elfin.wcn.retrofitJs",
                    // 菜单排序
                    "group": "navigation"
                }
			]
		}
	},
}
```

### extension.js
```js
module.exports = function retrofitJs(context) {
	// 注册命令
	const disposable = vscode.commands.registerCommand('elfin.wcn.retrofitJs', () => {});

	context.subscriptions.push(disposable);
};
```
