```js
{
    "activationEvents": [
        // vscode.languages 注册provider
		"onLanguage:javascript"

```

### extension.js
```js
/**
 * 查找文件定义的provider，匹配到了就return一个location，否则不做处理
 * 最终效果是，当按住Ctrl键时，如果return了一个location，字符串就会变成一个可以点击的链接，否则无任何效果
 * @param {*} document 
 * @param {*} position 
 * @param {*} token 
 */
function provideDefinition(document, position, token) {
    // new vscode.Position(0, 0) 表示跳转到某个文件的第一行第一列
    return new vscode.Location(vscode.Uri.file(''), new vscode.Position(0, 0));
}

module.exports = function (context) {
    // 注册如何实现跳转到定义，第一个参数表示仅对javascript文件生效
    context.subscriptions.push(vscode.languages.registerDefinitionProvider(['javascript'], {
        provideDefinition
    }));
};
```
